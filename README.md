# 安装

npm i | yarn

# 变量配置

.evn 中配置

MongoDB 数据库

Redis 缓存

# docker 启动 mongo 数据库 和 redis，可以使用本地数据库

docker-compose up -d

# pm2 启动

pm2 start ecosystem.config.js

# 直接启动

node app.js
